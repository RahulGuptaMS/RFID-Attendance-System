#include <SPI.h>
#include <MFRC522.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// RFID Pins
#define SS_PIN 10
#define RST_PIN 9
MFRC522 rfid(SS_PIN, RST_PIN);

// LCD (Change address if needed 0x27 / 0x3F)
LiquidCrystal_I2C lcd(0x27, 16, 2);

// LED & Buzzer
#define GREEN_LED 6
#define RED_LED 7
#define BUZZER 5

// Your Card UIDs
String uid1 = "DBCA106";   // Rahul & Uttam
String uid2 = "13BD354";   // Deepanshu & Komal

void setup() {
  Serial.begin(9600);
  SPI.begin();
  rfid.PCD_Init();

  pinMode(GREEN_LED, OUTPUT);
  pinMode(RED_LED, OUTPUT);
  pinMode(BUZZER, OUTPUT);

  lcd.init();
  lcd.backlight();

  lcd.setCursor(0, 0);
  lcd.print("Scan Card...");
}

void loop() {

  // Wait for card
  if (!rfid.PICC_IsNewCardPresent()) return;
  if (!rfid.PICC_ReadCardSerial()) return;

  String uid = "";

  // Read UID
  for (byte i = 0; i < rfid.uid.size; i++) {
    uid += String(rfid.uid.uidByte[i], HEX);
  }

  uid.toUpperCase();
  Serial.println(uid);

  lcd.clear();

  // ✅ Card 1
  if (uid == uid1) {
    lcd.setCursor(0, 0);
    lcd.print("Attendance OK");
    lcd.setCursor(0, 1);
    lcd.print("Rahul & Uttam");

    digitalWrite(GREEN_LED, HIGH);
    delay(2000);
    digitalWrite(GREEN_LED, LOW);
  }

  // ✅ Card 2
  else if (uid == uid2) {
    lcd.setCursor(0, 0);
    lcd.print("Attendance OK");
    lcd.setCursor(0, 1);
    lcd.print("Deepanshu Kom");

    digitalWrite(GREEN_LED, HIGH);
    delay(2000);
    digitalWrite(GREEN_LED, LOW);
  }

  // ❌ Wrong Card
  else {
    lcd.setCursor(0, 0);
    lcd.print("Access Denied");

    digitalWrite(RED_LED, HIGH);
    digitalWrite(BUZZER, HIGH);

    delay(2000);

    digitalWrite(RED_LED, LOW);
    digitalWrite(BUZZER, LOW);
  }

  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Scan Card...");
}
